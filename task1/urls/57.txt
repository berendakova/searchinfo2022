    $(document).ready(function(){
        var contentLang = 'ru';
        var storageLang = localStorage.getItem('contentLang');
        var dontShowChangeLang = localStorage.getItem('dontShowChangeLang');

        $('#checked-change-lang-content').click(function(){
            localStorage.setItem("dontShowChangeLang", 1);
            $('#modal-change-lang-content').modal('hide');
        });

        if(!storageLang) {
            localStorage.setItem("contentLang", contentLang);
        } else {
            dontShowChangeLang = !dontShowChangeLang || dontShowChangeLang == 0;

            if (storageLang && contentLang && storageLang.length === 2 && contentLang !== storageLang && dontShowChangeLang ) {
                $.ajax({
                    type: 'GET',
                    url: '/site/get-translate-by-langs',
                    data: {
                        withLang: storageLang,
                        toLang: contentLang
                    },
                    dataType: 'json',
                    success: function (data) {
                        localStorage.setItem("contentLang", contentLang);
                        $('#textWith').text(data.textWith);
                        $('#textTo').text(data.textTo);
                        $('#modal-change-lang-content').modal();
                    }
                });
            }
        }
    });
</script>
